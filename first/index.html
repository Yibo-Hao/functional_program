# 第一章-函数式编程概览

# 说在前面

我们平常写代码通常还是命令式编程，面向对象以及函数式更多的用在库的源代码中，在库的约束下我们使用命令式编程来写业务，我们使用类不能说我们使用了面向对象，只能说我们使用了类的语法。

我们学习的只是函数式的语法，但是函数式的思想需要我们慢慢消化

# 函数式的编程

函数式就是利用纯函数来实现一些细粒度函数，然后通过函数组合来把这些细粒度函数组合成功能强大的函数。

React 以及 Vue3 使用了函数式编程。

函数式编程是一种编程思想，和面向对象，面向过程并列。

函数式编程的思维方式：把现实事物之间的联系抽象到程序中。面向对象是对事物本身进行抽象。值得注意的是这里的函数值得是数学中的函数。数学中的函数是用来描述映射关系的，有以下特点，相同的输入始终得到相同的输出。

函数式编程就是用来描述数据之间的映射，过程的抽象。简称FP。

//非函数式
let num1 = 1
let num2 = 1
let sum = num1 + num2
//函数式
function add (n1,n2){return n1 + n2}

# 函数是一等公民（first class citizen）

函数在JS中是一等公民，具体在下面几个方面。

  • 函数可以存储在变量中
  • 函数作为参数
  • 函数作为返回值

# 高阶函数

  • 函数作为参数
  • 函数作为返回值
function forEach(array,fn) {
    for (let i = 0;i<array.length;i++){
        fn(array[i])   
    }
}

function filter(array,fn) {
    let results = []
    for (let i = 0;i<array.length;i++){
        if (fn(array[i])){
            results.push(array[i])    
        }    
    }
    return results
}

让函数只能执行一次的函数，arguments代表实参，虽然pay函数在定义时没有参数，但是只要传递实参即可。

function once(fn) {
    let done = false
    return function() {
        if (!done){
            done = true
            return fn.apply(this,arguments)
        }
    }
}
const pay = once(function (money) {
    console.log(money);
})
pay(1)

# 模拟数组的高阶函数

function map(array,fn) {
    let results = []
    for (let i = 0;i<array.length;i++){
        results.push(fn(array[i]))   
    }
    return results
}

function every(array,fn) {
    let result = true
    for (let i = 0;i<array.length;i++){
        result = fn(value)   
        if (!result){
            break    
        }
    }
    return result
}

const some = (array,fn)=>{
    let result = false
    for (let i = 0;i<array.length;i++){
        result = fn(value)   
        if (result){
            break    
        }
    }
    return result
}

数组的高阶函数一般是对循环和查询的抽象。

# 细粒度

需求：对不同级别的员工按照不同的绩效来发工资

function makeSalary(base) {
  return function(performance) {
    return base + performance
  }
}
let salaryLevel1 = makeSalary(10000)
let salaryLevel2 = makeSalary(15000)
